Algorithme du TP2
La fonction estValide()

Variables :
boolean grille[][];
boolean vérifierSelonColonne;
boolean dessinValide = true;
boolean dessinDébuté = false;
boolean contientCase = false;
boolean dessinTerminé = false;

Algorithme :

int i= 0;
TANT QUE (i < grille.length && dessinValide)
{
	contientCase= false;
	
	int j= 0;
	TANT QUE (j < grille[i].length && !contientCase)
	{
		if (VérifierCaseSelon(vérifierSelonColonne, grille, i, j))
		{
			contientCase = true;
		}
		
		++j;
	}
	
	if (contientCase && !dessinDébuté) 
	{
		dessinDébuté = true;
	}
	
	if (dessinDébuté && !contientCase) 
	{
		dessinTerminé = true;
		dessinDébuté = false;
	}
	
	if (dessinTerminé && contientCase)
	{
		dessinValide = false;
	}
	
	++i;
}


boolean VérifierCaseSelon(boolean vérifierSelonColonne, boolean[][] grille, int i, int j) {
	return (vérifierSelonColonne) ? grille[i][j]: grille[j][i];
}
